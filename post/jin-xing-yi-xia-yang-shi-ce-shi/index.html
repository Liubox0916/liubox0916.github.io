#include<iostream>
using namespace std;
int main(){
	int x,y,k;
	cin>>x>>y>>k;
	int ai,xi,bi,yi;
	int count=0,flag=0,ans=0;
	if(x==y){
		cout<<0;
		return 0;
	}
	while(k--){
		cin>>ai>>xi>>bi>>yi;	
		if(ai==1){
			x+=xi;
		}
		else if(ai==-1){
			x-=xi;
			if(x<0){
				x=0;
			}
			}
			if(bi==1){
				y+=yi;
			}
			else if(bi==-1){
				y-=yi;
				if(y<0){
				y=0;
				}
			}
			count++;
			if (x==y&&!flag){
				ans=count;
				flag=1;
			}
		} 
		if(flag){
			cout<<ans;
		}
		else{
			cout<<-1;	
		}	
	return 0;
}
